 -- 1. إنشاء جدول المستخدمين (أطباء ومرضى)
CREATE TABLE users (
    user_id SERIAL PRIMARY KEY,
    full_name VARCHAR(100) NOT NULL,
    email VARCHAR(100) UNIQUE NOT NULL,
    password_hash TEXT NOT NULL,
    role VARCHAR(20) ,
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 2. إنشاء جدول الملفات الشخصية للمرضى (بيانات ثابتة)
CREATE TABLE patient_profiles (
    profile_id SERIAL PRIMARY KEY,
    user_id INT REFERENCES users(user_id) ON DELETE CASCADE,
    birth_date DATE,
    gender VARCHAR(10),
    blood_type VARCHAR(5),
    chronic_diseases TEXT, -- الأمراض المزمنة إن وجدت
    updated_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 3. إنشاء جدول التحاليل الدورية (مصدر بيانات المخططات البيانية)
CREATE TABLE lab_results (
    result_id SERIAL PRIMARY KEY,
    patient_id INT REFERENCES users(user_id) ON DELETE CASCADE,
    test_name VARCHAR(50) NOT NULL, -- مثل: Glucose, Iron, Hemoglobin
    test_value NUMERIC(10, 2) NOT NULL, -- القيمة الرقمية للتحليل
    unit VARCHAR(20), -- وحدة القياس مثل mg/dL
    test_date DATE NOT NULL DEFAULT CURRENT_TIMESTAMP
);

-- 4. إنشاء جدول التقارير الطبية المشفرة
CREATE TABLE medical_reports (
    report_id SERIAL PRIMARY KEY,
    patient_id INT REFERENCES users(user_id) ON DELETE CASCADE,
    doctor_id INT REFERENCES users(user_id),
    report_content TEXT NOT NULL, -- هنا يتم تخزين النص المشفر (Encrypted Data)
    encryption_key_id VARCHAR(100), -- مرجع لمفتاح التشفير إذا كنت تستخدم نظام إدارة مفاتيح
    created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP
);

-- 5. إنشاء فهارس (Indexes) لتسريع البحث (مهم جداً للمخططات البيانية)
CREATE INDEX idx_lab_results_patient_date ON lab_results(patient_id, test_date);


 CREATE TABLE vital_signs (
    sign_id SERIAL PRIMARY KEY,
    patient_id INT REFERENCES users(user_id) ON DELETE CASCADE,
    blood_pressure_sys INT, -- ضغط الدم الانقباضي (مثل 120)
    blood_pressure_dia INT, -- ضغط الدم الانبساطي (مثل 80)
    heart_rate INT,         -- معدل ضربات القلب (نبضة/دقيقة)
    blood_sugar NUMERIC(5,2),-- سكر الدم (mg/dL)
    oxygen_level INT,       -- تشبع الأكسجين (%)
    measured_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP -- وقت القياس بدقة
);

 CREATE TABLE doctor_patient_assignment (
    assignment_id SERIAL PRIMARY KEY,
    doctor_id INT REFERENCES users(user_id) ON DELETE CASCADE,
    patient_id INT REFERENCES users(user_id) ON DELETE CASCADE,
    assigned_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    status VARCHAR(20) DEFAULT 'active', -- حالة المتابعة (نشط/مكتمل)
    UNIQUE(doctor_id, patient_id) -- لمنع التكرار لنفس الطبيب والمريض
);

DELIMITER $$

CREATE TRIGGER auto_create_profile
AFTER INSERT ON users
FOR EACH ROW
BEGIN
    -- Patient
    IF NEW.role = 'patient' THEN
        INSERT INTO patient_profiles (user_id)
        VALUES (NEW.user_id);
    END IF;

    -- Doctor
    IF NEW.role = 'doctor' THEN
        INSERT INTO doctor_profiles (user_id, medical_id, specialization)
        VALUES (NEW.user_id, NULL, NULL);
    END IF;

    -- Nurse
    IF NEW.role = 'nurse' THEN
        INSERT INTO nurse_profiles (user_id, department, shift)
        VALUES (NEW.user_id, NULL, NULL);
    END IF;
END$$

DELIMITER ;
